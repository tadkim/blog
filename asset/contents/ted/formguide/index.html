1차 형태 제작 가이드

Mike Bostock의 Superformular Explorer에서 10가지 형태를 채택하고 어떤 직군에 부여할지, 어떤 값들을 조절하여 형태를 변화시킬지, 최대최소값은 얼마인지 등을 전달하는 가이드이다. 코드, 이미지, 설명 등을 첨부하였다. 특정 형태가 어떤 Role에 적용될 지는 공평하게 사다리타기로 결정하였다.

전체 코드

예제에서 변하는 값은 m,n1,n2,n3,a,b가 있다. 변하는 값이 두 가지일 경우 종속적인 값은 2차 방정식으로 제시하였다.


var types = {
  art: {m:, n1:, n2:, n3:, a:, b:},
  explorer: {m:1.362, n1:1.362, n2:9.237, n3:5.938, a:-0.9625, b:1.025},
    //b, 1, 1.475
    //n2, (226*b-29)/33, (140*b+31)/19
  humanist: {m:, n1:, n2:, n3:, a:, b:},
  management: {m:-2.537, n1:407.1, n2:603.1, n3:936.3, a:1.287, b:0.5375},
    //
    //
  media: {m:, n1:, n2:, n3:, a:, b:},
  profession: {m:4.000, n1:632.5, n2:421.8, n3:100.0, a:2.000, b:1.000},
    //n1, 300, 670
    //n2, (8*n1+9440)/37, (5*n1+18850)/37
  public: {m:, n1:, n2:, n3:, a:, b:},
  religion: {m:, n1:, n2:, n3:, a:, b:},
  science: {m:, n1:, n2:, n3:, a:, b:},
  vip: {m:, n1:, n2:, n3:, a:, b:}
};
